# Connect using private key
ssh -i ~/.ssh/key root@localhost -p 2222



# files for the current user ~/.ssh
known_hosts             - Identitaten von Remote Servern, die User kennt und verbinden kann (*key.pub)
.ssh/authorized_keys    - Public Keys von den Clients, die sich verbinden können
# directory und file müssen ausfürbar gemacht werden
chmod 700 .ssh && chmod 600 .ssh/authorized_keys

# files die für alle User gelten im system /etc/ssh
ssh_known_hosts         - Identitaten von Remote Servern, die alle User kennen und verbinden können (*key.pub)
ssh_config              - Config file für den SSH Dienst
# StrictHostKeyChecking kann auf Yes gestellt werden. Wird abgebrochen wenn keys nicht mehr übereinstimmen
# Password-Atheitication kann auf no gestellt werden um keine Passwörter mehr zuzulassen
