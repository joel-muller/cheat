# verify the jenv installation
jenv doctor

# enable the export plugin (restart the shell after this)
jenv enable-plugin export

# add jdks
jenv add /System/Library/Java/JavaVirtualMachines/1.6.0.jdk/Contents/Home

# list managed jdks
jenv versions

# set new global java version
jenv global oracle64-1.6.0.39
jenv global 25

# check version
java --version
